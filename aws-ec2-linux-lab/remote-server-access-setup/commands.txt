# EC2 Remote Access Commands

# Navigate to directory containing key pair (Linux/WSL)
cd ~/Downloads

# Set correct permissions for key file
chmod 400 ec2-keypairs.pem


# Connect to EC2 Instance via SSH

ssh -i ec2-keypairs.pem ubuntu@<public-ip>


# Basic Verification Commands (after login)

whoami
hostname
pwd
ls -la


# Update System Packages

sudo apt update


# Exit SSH Session

exit
